# [Installation steps]
#
# Step 1: PyTorch (Jetson-specific build; standard pip does not support ARM64+CUDA)
#   pip install --no-cache \
#     "https://developer.download.nvidia.com/compute/redist/jp/v61/pytorch/\
#      torch-2.5.0a0+872d972e41.nv24.08.17622132-cp310-cp310-linux_aarch64.whl"
#
#   For torchaudio, match the PyTorch version using:
#     https://forums.developer.nvidia.com/t/pytorch-for-jetson/72048
#
# Step 2: CTranslate2 (faster-whisper dependency; pre-built in this environment)
#   pip install /home/tirobot/whisper/CTranslate2/python/dist/*.whl
#
#   If no pre-built whl is available, install via NVIDIA index:
#     pip install ctranslate2 --extra-index-url https://pypi.ngc.nvidia.com
#
# Step 3: Packages in this file
#   pip install -r requirements.txt
#
# Step 4: System packages (required by soundfile)
#   sudo apt install libsndfile1
#
# ============================================================
